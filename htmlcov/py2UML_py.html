38 def select_files(self): 

39 b_list = [] 

40 for b in self.black_list: 

41 b_list += (glob(f'{self.in_path}/{b}*', recursive=True)) 

42 if ".py" in self.in_path: 

43 return [self.in_path] 

44 w_list = glob(f'{self.in_path}\**\*.py', recursive=True) 

45 

46 new_b_list = [] 

47 for selected_item in w_list: 

48 for unselected_item in b_list: 

49 if selected_item in unselected_item:
